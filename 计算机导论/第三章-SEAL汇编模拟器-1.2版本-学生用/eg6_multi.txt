#<汇编代码：两个数的乘法>
move R0,4      #a = 4
move R2,8      #b = 8
move R1,0      #设置最终结果的初值为0
L1:            
and R3,R2,1     
beqz R3,L2     
add R1,R1,R0   #使用寄存器R4存储每次做加法后的值，计算结束时为最终结果
L2:            
shiftr R2,R2,1   
shiftl R0,R0,1    
sle R4,R2,0     
beqz R4,L1    
_pr R1
